SELECT ?REVENUE ?CITY ?REGION ?COUNTRY 
WHERE {
  VALUES ( ?REVENUE ?CITY ?REGION ?COUNTRY ) { ( s:REVENUE s:CITY s:REGION s:COUNTRY ) }
  s:Sales G:hasFeature s:REVENUE .
  s:Sales s:location s:City .
  s:City G:hasFeature s:CITY .
  s:City s:partOf s:Region .
  s:Region G:hasFeature s:REGION .
  s:Region s:partOf s:Country .
  s:Country G:hasFeature s:COUNTRY .
  
}
